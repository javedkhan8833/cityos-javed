cat << 'EOF' > deploy-run.sh
#!/bin/bash
echo "=== Production Startup ==="

cd /home/runner/workspace/api

# Ensure directories exist
mkdir -p storage/logs storage/app/public storage/framework/{cache,sessions,testing,views}

# Clear all caches to ensure fresh config
php artisan config:clear 2>&1 || true
php artisan route:clear 2>&1 || true
php artisan view:clear 2>&1 || true

# Run DB tasks in background with a longer delay to stay out of the way of the platform sync
(
    sleep 15
    echo "Running background database checks..."
    
    # Try to install extensions silently
    psql "$DATABASE_URL" -c "CREATE EXTENSION IF NOT EXISTS pg_trgm;" 2>/dev/null || true
    
    # Only run migrations if we really need to
    # This prevents the 'Duplicate table' error from blocking anything
    php artisan migrate --force --no-interaction 2>&1 || echo "Migrations skipped or already up to date"
    
    echo "Background tasks complete"
) &

echo "Starting PHP server on 0.0.0.0:5000..."
exec php -d memory_limit=1024M -S 0.0.0.0:5000 -t public server.php
EOF
